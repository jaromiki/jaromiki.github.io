Šablóny - Templates

Šablóny (generické funkcie)nám umožňujú zadefinovať funkcie do ktorých prenesieme typ premennej/premenných. Napríklad triedenie quicksort je rovaké pre prvky celočíselné, reálne prípadne aj pre textové. Myšlienka je podobná ako pri preťažovaní funkcií.
Generické funkcie vytvárame pomocou kľúčového slova template
Unit1.cpp
#include 
using namespace std;

template <class X>      //X je názov premenného typu
X vacsi(X a, X b)               //definícia funkcie typu X (bude určený pri volaní)
{   return a>b?a:b;}

int main()
{   int i=2, j=3;
    float x=1.5, y=5;
    cout <<vacsi(i,j) << endl;
    cout <<vacsi(x,y) << endl;
    cout <<vacsi(10,x) << endl;     //nesprávne parametre musia byť rovnakého typu
	
    cin.get();
    return 0;
}
Funkcia sa pri volaní pozrie aké sú typy parametrov a podľa toho určí typ X. Namiesto kľúčového slova class možno použiť aj typename
template < typename X>

Ak potrebujeme môžme do funkcie doniesť aj dva typy.
Unit1.cpp
#include 
using namespace std;

template <class X>      //X je názov premenného typu
void vymen(X &a, X &b)               //parametre rovnakého typu
{   X c=a; a=b; b=c;
    return a>b?a:b;}
template <class X, class Y>      //X je názov premenného typu
void vymen(X &a, Y &b)               //parametre nerovnakého typu
{   X c=a; a=b; b=c;
    return a>b?a:b;}

int main()
{   int i=2, j=3;
    float x=1.5, y=5;
	vymen(i,j);             //zavola vymen s rovnakými parametrami
    cout << "i=" << i <<", j="<<j <<endl;
    vymen (i,y);            //zavola vymen s nerovnakými parametrami 
    cout << "i=" << i <<", y="<<y <<endl;
    cin.get();
    return 0;
}

Unit1.cpp
#include

class zasobnik
{   X x[100];
