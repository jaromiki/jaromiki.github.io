Unit1.cpp
#incluse <iostream>
using namespace std;
class priklad
{  int a;
   public:
   void set_a(int aa) { a = aa;}
   int get_a() {return a;}
};
int main ()
{  priklad x[5];    //vztvorí 5 statickıch tried
   for (int i = 0; i<5; i++) x[i].a = i;
   for (int i = 0; i<5; i++) cout << "trieda:" << i<<" : "<< x[i].a;
   cout <<endl;
   return 0;
}

Inicializácia triedy v poli pomocou konštruktora

Ak by sme v triede priklad zadali aj konštruktor, mohli by triedy inicializovaná pri deklarácii.
Unit1.cpp
#include <iostream>
using namespace std;
class priklad1
{  int a;
   public:
   void set_a(int aa) { a = aa;}
   int get_a() {return a;}
   priklad1(int aa) {a = aa;}
   priklad1() {a = 100;}
};
class priklad2
{  int a,b;
   public:
   int get_a() {return a;}
   int get_b() {return b;}
   priklad2(int aa, int bb) {a = aa;b=bb;}
};

int main ()
{  priklad1 x[5]={10, 20, 30, 40, 50};
   priklad2 y[5]={priklad2(10,15), priklad2(10,15), priklad2(10,15), priklad2(10,15), priklad2(10,15)};
   
   return 0;
}
V pripade ak má trieda v konštruktore zadanı len jeden údaj mono poui skrátenı inicializaènı zápis V prípade ak má trieda v konštruktore zadanıch viac údajov musíme poui úplná inicializaènı zápis

Inicializácia triedy v poli vytváranej pomocou new

V predošlom príklade ak by som vytváral triedy príklad1 a príklad2 dynamicky bolo by to nasledovné: 
int main ()
{  priklad1 *x =new priklad1[5];   //vytvori 5 prvkove pole tried, kde v kazdej triede nastavi a = 100
   priklad1 *c =new priklad1(5);   //vytvori 1 prvkove pole tried, kde v triede nastavi a = 5
   ...
   delete [] x;
   delete c;
}
