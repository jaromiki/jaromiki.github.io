Príklady 
	// *********************  Hodnota  *****************************
		var n = 1; 		// premenna n obsahuje hodnotu 1
		var m = n;		// kopirovanie hodnotou, v m je ina hodnota 1
		
		function pripocitaj(spolu, x)
		{ spolu += x; }
		
		// volanie funkcie
		pripocitaj (n,m);		// premenne n,m sa nezmenia, funkcia vlastne nerobi nic
		alert(n == m);			// vypíše true
		
	// *********************  Odkaz  *****************************

		var slnovrat = vianoce; 		// obe premenne su na tej istej adrese
		
		slnovrat.setDate(21); 			// zmeni den v premennej slnovrat
		alert(vianoce.getDate();		// vypise hodnotu 21
